Exploration of the problem

Approach via SDP using piecewise polynomials

First produce ω and its pieces
k = 4; % omega is of degree n=k-1
t = sort(2*rand(1,k-1)-1);
x = chebfun('x');
omega = 1/factorial(k-1);
for l=1:k-1
omega = omega*(x-t(l));
end
tt = [-1, t, 1];
omegaPP = cell(k,k); % smoothness, then subinterval
for s=0:k-1
for l=1:k
omegaPP{s+1,l} = diff( chebfun(omega,[tt(l),tt(l+1)]) ,s);
end
end
Then create the "change-of-basis" and "smoothness-condition" matrices
N = 7; % degree of each piece of Omega
% change-of-basis
M = zeros(N+1,N+1,k,k);
for l=1:k
for s=0:k-1
for n=0:N
M(1:n+1,n+1,s+1,l) = ...
chebcoeffs( (-1)^(k-s-1) * diff(chebpoly(n,[tt(l),tt(l+1)]),k-s-1) * omegaPP{s+1,l} );
end
end
end
% smoothness-condition
Dleft = zeros(k-1,N+1,k);
Dright = zeros(k-1,N+1,k);
for l=1:k
for s=0:k-2
for n=0:N
aux = diff( chebpoly(n,[tt(l),tt(l+1)]) , s);
Dleft(s+1,n+1,l) = aux(tt(l));
Dright(s+1,n+1,l) = aux(tt(l+1));
end
end
end
Solve the semidefinite program and create Ω
cvx_begin quiet
variable d;
variable a(N+1,k);
variable Q(N+1,N+1,k,k) hermitian semidefinite; % smoothness, then subinterval
minimize d
subject to
sum(sum(a)) == 1; % to avoid positive multiples of Omega
% nonnegativity conditions
for l=1:k
for s=0:k-1
%n=0
sum(diag(Q(:,:,s+1,l))) == M(1,:,s+1,l)*a(:,l) + d;
%n=1:N
for n=1:N
sum(diag(Q(:,:,s+1,l),-n)) == M(n+1,:,s+1,l)*a(:,l)/2;
end
end
end
% smoothness conditions
Dleft(:,:,1)*a(:,1) == 0;
for l=1:k-1
Dright(:,:,l)*a(:,l) == Dleft(:,:,l+1)*a(:,l+1);
end
Dright(:,:,k)*a(:,k) == 0;
cvx_end
% return the output
pieces = cell(k,k); % smoothness, then subinterval
for l=1:k
aux = chebfun(a(:,l),[tt(l),tt(l+1)],'coeffs');
for s=0:k-1
pieces{s+1,l} = diff(aux,s);
end
end
OmegaPP_diffs = cell(k,1);
for s=0:k-1
OmegaPP_diffs{s+1} = chebfun(pieces(s+1,:),tt);
end
cvx_optval
cvx_optval = 1.0394e-07
