Construction of the functions Ω (normalized to have integral one) for each of our three options.

PW1 = cell(n+1,1);
PW2 = cell(n+1,1);
PW3 = cell(n+1,1);
PW1{1} = (-1)^n*chebfun('1',[t(1),t(2)]);
PW2{1} = chebfun(p,[t(1),t(2)]);
PW3{1} = chebfun(p*abs(p),[t(1),t(2)]);
for j=1:n
PW1{j+1} = a1(j)*(-1)^(n-j)*chebfun('1',[t(j+1),t(j+2)]);
PW2{j+1} = a2(j)*chebfun(p,[t(j+1),t(j+2)]);
PW3{j+1} = a3(j)*chebfun(p*abs(p),[t(j+1),t(j+2)]);
end
Omega1 = cumsum((-1)^n*chebfun(PW1,t),n);
Omega1 = Omega1/sum(Omega1);
Omega2 = cumsum((-1)^n*chebfun(PW2,t),n);
Omega2 = Omega2/sum(Omega2);
Omega3 = cumsum((-1)^n*chebfun(PW3,t),n);
Omega3 = Omega3/sum(Omega3);
