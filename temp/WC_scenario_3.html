The code for the construction of the function f

function [f,df] = WC_function(X,Y)
 
x = chebfun('x');
 
% set an solve the linear system M*a = -b
n = length(X);
M = zeros(n-1,n-1);
b = zeros(n-1,1);
for i=1:n-1
T = chebpoly(i-1);
for j=1:n-1
M(i,j) ...
= sum((x-Y(j))*T,[Y(j),X(j)])/(X(j)-Y(j))...
+ sum((Y(j+1)-x)*T,[X(j),Y(j+1)])/(Y(j+1)-X(j));
end
b(i) ...
= sum((x-Y(n))*T,[Y(n),X(n)])/(X(n)-Y(n))...
+ sum((Y(n+1)-x)*T,[X(n),Y(n+1)])/(Y(n+1)-X(n));
end
a = [-M\b;1];
 
% construct the function f and its derivative df via their pieces
pieces_f = cell(2*n,1);
pieces_df = cell(2*n,1);
for j=1:n
pieces_f{2*j-1} = a(j)*chebfun(x-Y(j),[Y(j),X(j)])/(X(j)-Y(j));
pieces_df{2*j-1} = a(j)*chebfun(1,[Y(j),X(j)])/(X(j)-Y(j));
pieces_f{2*j} = a(j)*chebfun(Y(j+1)-x,[X(j),Y(j+1)])/(Y(j+1)-X(j));
pieces_df{2*j} = -a(j)*chebfun(1,[X(j),Y(j+1)])/(Y(j+1)-X(j));
end
XY = sort([X,Y]);
f = chebfun(pieces_f,XY);
df = chebfun(pieces_df,XY);
 
end
